COVISE_FIND_PACKAGE(NetCDF)

IF (NOT NETCDF_FOUND)
  RETURN()
ENDIF (NOT NETCDF_FOUND)

INCLUDE_DIRECTORIES(
    ${NETCDF_INCLUDE_DIRS}
)

SET(HEADERS
)
SET(SOURCES
  ReadMeteoCosmo.cpp
)
ADD_COVISE_MODULE(IO ReadMeteoCosmo ${EXTRASOURCES})
TARGET_LINK_LIBRARIES(ReadMeteoCosmo coApi coAppl coCore ${NETCDF_LIBRARIES} ${EXTRA_LIBS})
COVISE_INSTALL_TARGET(ReadMeteoCosmo)
